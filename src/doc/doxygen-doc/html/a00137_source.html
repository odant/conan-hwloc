292 static __hwloc_inline int
293 hwloc_distances_obj_pair_values(struct hwloc_distances_s *distances,
294  hwloc_obj_t obj1, hwloc_obj_t obj2,
295  hwloc_uint64_t *value1to2, hwloc_uint64_t *value2to1)
296 {
297  int i1 = hwloc_distances_obj_index(distances, obj1);
298  int i2 = hwloc_distances_obj_index(distances, obj2);
299  if (i1 < 0 || i2 < 0)
300  return -1;
301  *value1to2 = distances->values[i1 * distances->nbobjs + i2];
302  *value2to1 = distances->values[i2 * distances->nbobjs + i1];
303  return 0;
304 }
